Ví dụ thực tế:

Code mẫu:

// Thành công
Toast.success('Đã thêm vào giỏ hàng!');

// Lỗi
Toast.error('Vui lòng điền đầy đủ thông tin!');

// Cảnh báo
Toast.warning('Số điện thoại phải có 10 chữ số!');

// Thông tin
Toast.info('Đang xử lý yêu cầu...');

// Tùy chỉnh thời gian (ms)
Toast.success('Thành công!', 5000);

// Hoặc dùng hàm chung
showToast('Thông báo', 'success');
